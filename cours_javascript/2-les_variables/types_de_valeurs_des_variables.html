La fonction typeof()

var b = "25" ;
var c = -25 ;
var d = 25.7 ;
var e = "Pierre" ;
var f = 'Bineng' ;
var g = "je parle deutsch" ;
var h = true ;
var i = false ;
var j = null ;
var k = undefined ;
var l = NaN ;
on applique la fonction typeof() pour déterminer le type de leurs valeurs : voir l'illustration #3
